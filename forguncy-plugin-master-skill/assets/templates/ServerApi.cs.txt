using System.Threading.Tasks;
using Microsoft.AspNetCore.Mvc;
using GrapeCity.Forguncy.ServerApi;

namespace MyPluginNamespace
{
    public class MyServerApi : ForguncyApi
    {
        [HttpGet]
        [Route("my-plugin/hello")]
        public async Task<IActionResult> Hello()
        {
            // 通过 this.Context 访问活字格上下文
            // var user = this.Context.User;
            
            return new JsonResult(new { message = "Hello from Server API" });
        }

        [HttpPost]
        [Route("my-plugin/data")]
        public async Task<IActionResult> ProcessData([FromBody] object data)
        {
            // 处理 POST 请求
            return Ok(new { status = "success", received = data });
        }
    }
}
